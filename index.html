const beerCardReducer = (state = initialState, action) => {
    switch (action.type) {
        case UPDATE_COMMENTS: {
            return {
                ...state,
                currentBeer: {
                    ...state.currentBeer,
                    reviews: action.reviews.length > 0 ? action.reviews : 
                    [...state.currentBeer.reviews]
                }
            };
        }
        default:
            return state;
    };
};

export default beerCardReducer;
export const updateComments = (reviews) => ({type: UPDATE_COMMENTS, 
					     reviews: reviews});
